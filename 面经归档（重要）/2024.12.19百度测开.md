void InsetSort(int a[], int n) {
	for(int i = 1; i < n; ++i) {
        if(a[i] < a[i-1]) {
            tmp = a[i];
            j = i;
            do {
                a[j] = a[j-1];
                j--;
            } while (j > 0 && a[j - 1] > tmp);
            a[j] = tmp;
        }
	}
}

// 交换排序
void Bubblesort(int a[], int n) {
    bool exchange;
    for(int i = 0; i < n - 1; ++i) {
        exchange = false;
        for(int j = n - 1; j > i; --j){
            if(a[j] < a[i]){
                swap(a[i], a[j]);
                exchange = true;
            }
            if(!exchange)return;
        }
    }
}

  将系统的输入域分为若干部分，从每个部分选取少量代表性数据进行测试。它认为某个等价类中一个值通过测试，那么在其它值上也会通过，适用于输入数据较多的情况。
